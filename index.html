FIRST-TIME SETUP — SUPABASE CONFIGURATION
STEP 1 — CREATE SUPABASE PROJECT
Go to supabase.com → New Project → copy your Project URL and anon public key from Project Settings → API.
STEP 2 — RUN THIS SQL IN SUPABASE (SQL Editor)
CREATE TABLE factions (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), name text, shortname text, color text, emoji text, credits int DEFAULT 1000, supplies int DEFAULT 100, fuel int DEFAULT 100, manpower int DEFAULT 500, description text, created_at timestamptz DEFAULT now());
CREATE TABLE planets (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), name text, sector text, emoji text, x float, y float, controlling_faction text DEFAULT 'Neutral', credit_yield int DEFAULT 50, supply_yield int DEFAULT 10, fuel_yield int DEFAULT 10, strategic_value text DEFAULT 'Low', description text);
CREATE TABLE planet_connections (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), planet_a text, planet_b text);
CREATE TABLE units (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), name text, faction text, unit_type text, size text, status text DEFAULT 'Active', planet_id uuid REFERENCES planets(id), strength int DEFAULT 100, notes text);
CREATE TABLE campaign_log (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), event_type text, description text, issued_by text DEFAULT 'Admin', created_at timestamptz DEFAULT now());
CREATE TABLE settings (key text PRIMARY KEY, value text);
INSERT INTO settings VALUES ('campaign_name','Star Wars Galaxy Campaign'),('campaign_era','Galactic Civil War'),('campaign_turn','1');

CREATE POLICY "public read" ON settings FOR SELECT USING (true);
CREATE POLICY "anon write" ON factions FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon write" ON planets FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon write" ON planet_connections FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon write" ON units FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon write" ON campaign_log FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "anon write" ON settings FOR ALL USING (true) WITH CHECK (true);
