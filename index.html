CREATE TABLE factions (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), name text UNIQUE, shortname text, color text, emoji text, credits int DEFAULT 1000, created_at timestamptz DEFAULT now());
CREATE TABLE planets (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), name text UNIQUE, sector text, emoji text, x float, y float, controlling_faction text DEFAULT 'Neutral', credit_yield int DEFAULT 150, strategic_value text DEFAULT 'Medium', description text);
CREATE TABLE planet_connections (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), planet_a text, planet_b text, UNIQUE(planet_a, planet_b));
CREATE TABLE units (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), name text, faction text, unit_type text, size text, status text DEFAULT 'Active', planet_id uuid REFERENCES planets(id) ON DELETE CASCADE, strength int DEFAULT 100);
CREATE TABLE campaign_log (id uuid PRIMARY KEY DEFAULT gen_random_uuid(), event_type text, description text, issued_by text DEFAULT 'Admin', created_at timestamptz DEFAULT now());
CREATE TABLE settings (key text PRIMARY KEY, value text);
